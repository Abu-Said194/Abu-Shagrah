function openFullscreenBubble(id) { const existing = document.querySelector(".fullscreen-bubble"); if (existing) existing.remove(); const bubbleDiv = document.querySelector(`.bubble[data-fullscreen-id="${id}"]`); if (!bubbleDiv) return; const fullscreen = document.createElement("div"); fullscreen.classList.add("fullscreen-bubble"); // Container für Text const textContainer = document.createElement("div"); textContainer.classList.add("text-content"); // Überschrift const h3 = document.createElement("h3"); h3.textContent = bubbleDiv.querySelector("h3").textContent; textContainer.appendChild(h3); // Text (Paragraph + Listen) kopieren bubbleDiv.querySelectorAll("p, ul").forEach(node => { textContainer.appendChild(node.cloneNode(true)); }); // Container für Bilder const imagesContainer = document.createElement("div"); imagesContainer.classList.add("images-content"); // Alle Bilder anzeigen if (bubblesWithImages[id]) { bubblesWithImages[id].forEach(src => { const img = document.createElement("img"); img.src = src; img.alt = `${id} Bild`; imagesContainer.appendChild(img); }); } fullscreen.appendChild(textContainer); fullscreen.appendChild(imagesContainer); // Schließen-Button const closeBtn = document.createElement("button"); closeBtn.textContent = "Schließen ✖"; closeBtn.classList.add("close-btn"); closeBtn.addEventListener("click", () => { fullscreen.remove(); }); fullscreen.appendChild(closeBtn); document.body.appendChild(fullscreen); // Scroll to top fullscreen.scrollTop = 0; }
